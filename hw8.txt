Не выполняя код, ответить, что выведет браузер и почему:

1.
if (!("a" in window)) {
    var a = 1;
}
alert(a);

блок if не будет исполнен, присвоения не будет ровно как и объявления, результатом - undefined.
	
2.
var b = function a(x) {
    x && a(--x);
};
alert(a);

Функциональных выражений не существует пока не наступит очередь переменных пахать, поэтому функцию интерпретатор не узнает.

3. function a(x) {
    return x * 2;
}
var a;
alert(a);

Сначала 'a' обозвали функцией, потом инициализировали как переменную, из-за этого в ней будет тело функции.

4.
function b(x, y, a) {
    arguments[2] = 10;
    alert(a);
}
b(1, 2, 3);
arguments - брат питоновского "*args", работает так же, локально внутри фукнции отдает аргументы. 
алерт выведет 10 потому что выше, третьему по счету аргументу - а, присваивают 10.

5.
function a() {
    alert(this);
}
a.call(null);

объект Window, потому что если передать null, вместо this в call() - он всегда вернет окошко.
